#!/bin/bash
SOURCE_DIR="/home/emerg1/sort"  # Change if needed
EMPTY_DIR="$SOURCE_DIR/empty_files"
NON_EMPTY_DIR="$SOURCE_DIR/non_empty_files"
# Create destination directories if they don't exist
mkdir -p "$EMPTY_DIR"
mkdir -p "$NON_EMPTY_DIR"
# Loop through files in the source directory
for file in "$SOURCE_DIR"/*.txt; do
        # Make sure it's a regular file
    if [ -f "$file" ]; then
            # skip if the file is in the empty or non empty folders already
            if [[ "$file" == "$EMPTY_DIR/"* || "$file" == "$NON_EMPTY_DIR/"* ]]; then
                    continue
            fi
         # -s checkes if the file is non-zero size
        if [ ! -s "$file" ]; then
            echo "Moving empty file: $file"
            mv "$file" "$EMPTY_DIR/"
        else
            echo "Moving non-empty file: $file"
            mv "$file" "$NON_EMPTY_DIR/"
        fi
    fi
done
echo "Sorting complete."
